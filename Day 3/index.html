DSA with JS - Day 3

Introduction

This document contains essential notes and code examples for learning Data Structures and Algorithms (DSA) using JavaScript. The topics covered include loops (for, while, do-while), break and continue statements, and practice problems like sum of digits, reverse of a number, strong numbers, and more.

For Loop

The for loop is used to iterate over a block of code a specific number of times.

Q1: Print "Hello World" N Times

for (let i = 0; i < 5; i++) {
    console.log("Hello World");
}

Q2: Sum of N Natural Numbers and Factorial of a Number

var number = Number(prompt("Enter your number: "));
var sum = 0;
for (let i = 1; i <= number; i++) {
    sum = sum + i;
}
var fact = 1;
for (var i = 1; i <= number; i++) {
    fact = fact * i;
}
console.log("The sum of n natural numbers is: ", sum);
console.log("The factorial is: ", fact);

Q3: Factors of a Number

for (var i = 1; i <= number; i++) {
    if (number % i === 0) {
        console.log(i);
    }
}

Q4: Prime Number

var primeIs = true;
for (var i = 2; i <= Math.floor(number / 2); i++) {
    if (number % i === 0) {
        primeIs = false;
        break;
    }
}
console.log(primeIs);

Break and Continue

for (let i = 10; i > 0; i--) {
    if (i === 8) {
        continue;
    } else {
        console.log(i);
    }
}

While Loop

The while loop is used to execute a block of code as long as a condition is true.

Q5: Sum of Digits (123)

var digit = 123;
let num = 0;
while (digit !== 0) {
    var newNum = digit % 10;
    num += newNum;
    digit = Math.floor(digit / 10);
}
console.log("The sum of digits", num);

Q6: Reverse of a Number

var reverseNumber = 123;
let reNum = 0;
while (reverseNumber !== 0) {
    var newNumber = reverseNumber % 10;
    reNum = reNum * 10 + newNumber;
    reverseNumber = Math.floor(reverseNumber / 10);
}
console.log("The reverse of the number is: ", reNum);

Q7: Strong Number

let strongNum = 145;
let originalNum = strongNum;
let sumfact = 0;
while (strongNum !== 0) {
    var newStrong = strongNum % 10;
    var factorial = 1;
    for (let i = 1; i <= newStrong; i++) {
        factorial *= i;
    }
    sumfact += factorial;
    strongNum = Math.floor(strongNum / 10);
}
if (sumfact === originalNum) {
    console.log("It is a strong number");
} else {
    console.log("It is not a strong number");
}

Do-While Loop

The do-while loop is used to execute a block of code at least once, and then repeat the loop as long as a condition is true.

Q8: Repeat "Hello" Until User Stops

var str_word;
do {
    str_word = prompt("If you want to stop, enter 0: ");
    console.log("Hello...");
} while (str_word !== "0");
